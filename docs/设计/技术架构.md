### 会话生命周期（发消息才建会话）

会话的创建遵循「**只要用户没发送消息就不应该产生对话 id**」原则，避免孤儿空会话：

1. **新建入口**：用户在 `/` 落地页输入框输入首条消息（或点击推荐主题 chip）→ 前端 `POST /api/sessions`（`topic="未命名对话"`）拿到真实 id → `navigate('/learn/:id', { state: { firstMessage } })`
2. **接续流**：`/learn/:id` 页接收 `firstMessage` 立即发起 chat 流（首条消息即触发 agent 诊断），messages 瞬间非空，无空态→hero→消息三态跳变（无闪烁）
3. **标题生成**：Worker 处理完首条消息后，异步调一次轻量模型生成 ≤10 字标题 → 更新 `session.topic` → 推 `title-updated` SSE 事件 → 前端左栏更新
4. **新对话按钮**：左栏「新对话」始终展示；0 对话时点击提示「当前已是新对话」（`/` 页即新对话入口，无需跳转），有对话时跳回 `/` 页

`POST /api/chat` 不再兜底创建会话——`sessionId` 不存在返回 404，暴露前端漏建为可排查错误。历史遗留的孤儿空会话（0 消息 active 会话）可通过 `pnpm --filter @ai-teacher/db cleanup-empty-sessions` 清理。

  system

  sending
  processing
  completed
  failed

enum SourceType {
  pdf

// 迭代 009：资料处理状态机，驱动 RAG 异步流水线 UI 反馈
enum SourceStatus {
  pending
  processing
  ready
  failed

model User {
  id         String          @id @default(cuid()) @db.Text
  name       String          @db.Text
  sessions   Session[]
  sources    Source[]
  profile    LearnerProfile?
  llmConfigs LlmConfig[]
  createdAt  DateTime        @default(now())
}

model Source {
  id        String       @id @default(cuid()) @db.Text
  userId    String       @db.Text
  user      User         @relation(fields: [userId], references: [id])
  title     String       @db.Text
  type      SourceType
  content   String?      @db.Text
  fileUrl   String?      @db.Text
  checksum  String?      @db.Text
  externalUrl String?    @db.Text  // 迭代 009 Phase 2：粘贴的原始视频链接（B站/YouTube/直链）
  status    SourceStatus @default(pending)
  sessions  Session[]
  chunks    DocumentChunk[]
  createdAt DateTime     @default(now())

  @@index([userId])

// 迭代 009：文档分块 + 向量存储（pgvector）。embedding 字段 Prisma 不原生支持，用 Unsupported 声明，retrieval service 经 raw SQL 检索；HNSW 索引由 migration raw SQL 创建
model DocumentChunk {
  id         String   @id @default(cuid()) @db.Text
  sourceId   String   @db.Text
  source     Source   @relation(fields: [sourceId], references: [id], onDelete: Cascade)
  chunkIndex Int
  content    String   @db.Text
  tokenCount Int
  embedding  Unsupported("vector(1024)")
  createdAt  DateTime @default(now())

  @@index([sourceId])
}

model Session {
  id          String    @id @default(cuid()) @db.Text
  userId      String    @db.Text
  user        User      @relation(fields: [userId], references: [id])
  topic       String    @db.Text
  sourceId    String?   @db.Text
  source      Source?   @relation(fields: [sourceId], references: [id])
  status      SessionStatus @default(active)
  summary     Json?
  teachingMode TeachingMode  @default(warm)
  activeMode   ActiveMode    @default(learning) // 迭代 051：当前学/固/验模式（spec §7）
  llmConfigId String?   @db.Text
  llmConfig   LlmConfig? @relation(fields: [llmConfigId], references: [id])
  roadmap     Roadmap?
  messages    Message[]
  checkpoints Checkpoint[]
  interviewResults InterviewResult[]
  createdAt   DateTime  @default(now())
  updatedAt   DateTime  @updatedAt

  @@index([userId, status])
  @@index([userId, updatedAt])
}

model Roadmap {
  id        String  @id @default(cuid()) @db.Text
  sessionId String  @unique @db.Text
  session   Session @relation(fields: [sessionId], references: [id])
  nodes     Node[]
  version   Int     @default(1)
}

model Node {
  id           String   @id @default(cuid()) @db.Text
  roadmapId    String   @db.Text
  roadmap      Roadmap  @relation(fields: [roadmapId], references: [id])
  index        Int
  title        String   @db.Text
  description  String   @db.Text
  status       NodeStatus @default(not_started)
  masteryScore Int      @default(0)
  reviewLog    Json?
  masteredAt   DateTime?
  // 迭代 051：间隔重复持久化（spec §3.4 / §7.1）。@default 兜底老数据，无需单独迁移脚本
  memoryStrength  Float     @default(1.0) // 老节点视为刚掌握、强度满
  lastReviewedAt  DateTime? // 未复习过
  nextReviewAt    DateTime? // null 表示逾期，进入复习模式优先呈现
  reviewInterval  Int       @default(1) // 当前复习间隔（天），首次 1d

  @@index([roadmapId, index])
  @@index([roadmapId, status])
}

// 迭代 052：面试结果（评分 + 难度状态 + 薄弱点 + 改进建议，spec §4/§7）
model InterviewResult {
  id          String              @id @default(cuid()) @db.Text
  sessionId   String              @db.Text
  session     Session             @relation(fields: [sessionId], references: [id])
  status      InterviewStatus     @default(in_progress)
  difficulty  InterviewDifficulty @default(medium) // 当前难度
  streak      Int                 @default(0) // 连续答对(+)/答错(-)

### 5.1 Agent Loop（run-agent-loop.ts）

```typescript
// 主 Agent 循环：手动多步迭代
// 每步调用 streamText(maxSteps=1) 执行一轮 tool calling
// 如果有工具调用 → 处理工具结果 → 继续下一轮
// 如果无工具调用 → 循环结束
async function runAgentLoop(params: AgentLoopParams) {
  for (let step = 0; step < MAX_STEPS; step++) {
    const result = await streamText({
      model: providerRegistry.getModel(llmConfig),
      system: buildSystemPrompt(promptSections), // learning→tutor / review→考官 / interview→面试官（迭代 051②/052②）
      messages: contextManager.buildMessages(session),
      tools: buildToolsForMode(session.activeMode), // learning→tutor / review→review / interview→interview 工具集
      maxSteps: 1, // 每步只执行一轮
    });

    // 流式推送 SSE 事件
    publishSSEEvents(result);

    // 无工具调用 → 结束循环
    if (!result.toolCalls?.length) break;

    // 处理工具结果
    await processToolResults(result.toolCalls);
  }
}
```

### 5.4 多 Agent 协作 + Subagent（迭代 022）

主 Agent（TutorAgent）可通过 `delegateTask` 工具将任务委派给专业子 Agent。每个子 Agent 有独立的 system prompt、工具集、模型和步数限制，执行结果经 `toModelOutput()` 过滤后返回摘要，防止上下文污染。

```typescript
// SubagentRegistry — apps/worker/src/agent/subagents/index.ts
interface SubagentDefinition {
  name: string;
  description: string;
  systemPrompt: string;
  tools: string[];
  maxSteps: number;
  model?: string; // 可指定模型
  toModelOutput: (result: AgentResult) => string;
}

// 注册的子 Agent
const registry = createSubagentRegistry();
// - assessment: 出练习题、评估答案、学习报告（maxSteps: 3, model: deepseek-v4-flash）
// - research: 搜索教学资料、补充参考（maxSteps: 5, model: deepseek-v4-flash）
```

```
TutorAgent (主)
  │
  ├── delegateTask({ agent: "assessment", task: "出 3 道 React Hooks 练习题" })
  │     └── AssessmentAgent (独立 streamText)
  │           ├── systemPrompt: 专业出题评估 Agent
  │           ├── tools: [assessMastery, generateAssessment]
  │           ├── maxSteps: 3
  │           └── toModelOutput → 摘要（≤300 字）
  │
  └── delegateTask({ agent: "research", task: "搜索 React Hooks 闭包陷阱资料" })
        └── ResearchAgent (独立 streamText)
              ├── systemPrompt: 只读研究 Agent
              ├── tools: [assessMastery]
              ├── maxSteps: 5
              └── toModelOutput → 摘要（≤500 字）
```

**关键设计**：

- `delegateTask` 通过工厂函数闭包捕获 `SubagentRegistry`，无需修改上下文
- 子 Agent 使用 `deepseek-v4-flash`（比主模型便宜），降低委派成本
- `toModelOutput()` 控制返回给主 Agent 的内容，过滤详细对话过程

## 5.4 A2UI 框架

AI-to-UI（A2UI）是消息渲染的抽象层，将 Agent 输出映射为结构化 UIBlock，前端通过 Registry 自动选择对应组件渲染。

### UIBlock 类型

| type             | 组件                        | 说明                                                    |
| ---------------- | --------------------------- | ------------------------------------------------------- |
| `text`           | TextBlock                   | Markdown 文本（含代码块、行内代码）                     |
| `assessment`     | AssessmentBlock             | 学习评估卡片（包装现有 AssessmentCard）                 |
| `quiz`           | QuizPanel                   | 交互式测验（可点击选项 + 即时反馈）                     |
| `code-result`    | CodeResultBlock             | 代码运行结果（代码 + stdout/stderr + 退出码）           |
| `formula`        | FormulaDisplay              | LaTeX 公式（KaTeX 渲染）                                |
| `diagram`        | DiagramRenderer             | 图表（预留占位）                                        |
| `table`          | TableBlock                  | 结构化表格（迭代 024）                                  |
| `callout`        | CalloutBlock                | 提示卡 tip/warning/key（迭代 024）                      |
| `comparison`     | ComparisonCard              | 对比卡 A vs B（迭代 024）                               |
| `heading`        | HeadingBlock                | 标题 h2/h3（迭代 029）                                  |
| `badge`          | BadgeBlock                  | 徽章标签 success/warning/info（迭代 029）               |
| `mastery-report` | MasteryReportBlock          | 掌握总结报告（迭代 029）                                |
| `interactive`    | InteractiveBlockRenderer    | 互动教学产物（自包含 HTML，iframe 沙箱渲染，迭代 050②） |
| `flashcard`      | FlashcardBlockRenderer      | 复习抽认卡（正面问题→翻面答案，迭代 051②）              |
| `interviewScore` | InterviewScoreBlockRenderer | 面试评分卡（总评+难度+薄弱点+改进建议，迭代 052②）      |

### 渲染流程

```
Message → 检查 metadata.uiBlocks
  ├── 有 uiBlocks → 遍历 blocks，Registry 查找组件渲染
  └── 无 uiBlocks → 降级为单个 TextBlock（向后兼容）
```
